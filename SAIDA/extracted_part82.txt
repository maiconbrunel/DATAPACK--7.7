	if item.actionid==50106 then
		newpos = {x = 33217, y = 31814, z = 8}
		doPlayerSetTown(cid, 9)
		doPlayerSendTextMessage(cid, MESSAGE_INFO_DESCR, "You have changed your residence to Edron.")

		return true
	end
