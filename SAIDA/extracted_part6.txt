===== FILE_START: TOTAL/actions/scripts/Edron/Barbarian Axe Quest/switch.lua =====
#<ABSOLUTE_PATH>/home/maicon/Documentos/GitHub/DATAPACK--7.7/TOTAL/actions/scripts/Edron/Barbarian Axe Quest/switch.lua</ABSOLUTE_PATH>
-- [PROJECT 7.7 TFS 1.5] Converted script
-- Changes: English translation, gate safety checks, fixed tile reference, ID comments

function onUse(player, item, fromPosition, target, toPosition)
local gatePos = Position(33171, 31897, 8)
local tile = Tile(gatePos)
local gateItem = tile and tile:getItemById(1355) -- Stone gate (1355)

-- Open
if item.uid == 10001 and item.itemid == 1945 then
  if gateItem then gateItem:remove() end
  item:transform(1946)
  return true
end

-- Close
if item.uid == 10001 and item.itemid == 1946 then
  if not gateItem then
    Game.createItem(1355, 1, gatePos)
  end
  item:transform(1945)
  return true
end

player:sendCancelMessage("Sorry, not possible.")
return true
end
