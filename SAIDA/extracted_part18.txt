-- [PROJECT 7.7 TFS 1.5] Converted script
-- Summary: Updated to player API, storage checks, messages to EN, safe addItem.

function onUse(player, item, fromPosition, target, toPosition)
	if item.uid ~= 10108 then
		return true
	end

	local status = player:getStorageValue(10108)

	if status <= 0 then
		player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "You have found a Talon.")
		player:addItem(2151, 1) -- Talon
		player:setStorageValue(10108, 1)
	else
		player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
	end

	return true
end
