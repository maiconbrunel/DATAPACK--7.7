-- [PROJECT 7.7 TFS 1.5] Converted script
-- Purpose: Teleport lever 1
-- Notes: API updates, safety checks

function onUse(player, item, fromPosition, target, toPosition)
	local dest = Position(368, 1386, 12)

	if item.uid == 10260 and item.itemid == 1945 then
		item:transform(1946)
		fromPosition:sendMagicEffect(2)
		player:teleportTo(dest)
		dest:sendMagicEffect(10)
		return true
	end

	if item.uid == 10260 and item.itemid == 1946 then
		item:transform(1945)
		return true
	end

	player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "Sorry, not possible.")
	return true
end
