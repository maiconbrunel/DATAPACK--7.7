-- Purpose: Triple UH chest
-- Notes: melhorias, API modernizada, tradução total
-- Changes: storage fix, english messages, API cleanup

function onUse(player, item, fromPosition, target, toPosition)
if item.uid ~= 1015 then
  return false
end

local storage = 1015
if player:getStorageValue(storage) < 1 and not player:getGroup():getAccess() then
  player:sendTextMessage(MESSAGE_INFO_DESCR, "You have found a triple ultimate healing rune.")
  player:addItem(2273, 3) -- Ultimate Healing Rune (2273)
  player:setStorageValue(storage, 1)
else
  player:sendTextMessage(MESSAGE_INFO_DESCR, "The chest is empty.")
end

return true
end

#<ABSOLUTE_PATH>/home/maicon/Documentos/GitHub/DATAPACK--7.7/TOTAL/actions/scripts/Edron/upgrade/upgrade - Copy.lua</ABSOLUTE_PATH>
-- [PROJECT 7.7 TFS 1.5] Converted script
-- Purpose: Upgrade system
-- Notes: API modernizada, correção crítica
-- Changes: toPos fix, english messages, variable conversion

function onUse(player, item, fromPosition, target, toPosition)
-- Validate anvil position
if toPosition.x == 65535 or not (toPosition.x == 33314 and toPosition.y == 31683 and toPosition.z == 11) then
  player:sendCancelMessage("Put the metal on the anvil first.")
  return true
end

-- 2258 → gold nugget
if target.itemid == 2258 then
  target:transform(2157) -- Gold Nugget (2157)
  toPosition:sendMagicEffect(CONST_ME_MAGIC_RED)
end

-- 2123 → royal steel (cost 100k)
if target.itemid == 2123 then
  if player:removeMoney(100000) then
    target:transform(2357) -- Royal Steel (2357)
    toPosition:sendMagicEffect(CONST_ME_MAGIC_RED)
  else
    player:sendCancelMessage("Not enough cash.")
  end
end

-- 2519 → upgraded via 100 gold nuggets
if target.itemid == 2519 then
  if player:getItemCount(2157) >= 100 then
    player:removeItem(2157, 100)
    target:transform(2539) -- Upgraded Gear (2539)
    toPosition:sendMagicEffect(CONST_ME_MAGIC_RED)
  else
    player:sendCancelMessage("Keep nuggets in your backpack.")
  end
end

return true
end
