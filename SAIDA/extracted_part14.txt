-- [PROJECT 7.7 TFS 1.5] Converted script
-- Purpose: Remove stone
-- Notes: API updates, safety checks

function onUse(player, item, fromPosition, target, toPosition)
	local gatePos = Position(434, 1326, 10)
	local tile = Tile(gatePos)
	local gate = tile and tile:getItemById(1304)

	if item.uid == 10258 and item.itemid == 1945 and gate then
		gate:remove()
		item:transform(1946)
		return true
	end

	if item.itemid == 1946 then
		item:transform(1945)
		return true
	end

	player:sendCancelMessage("Sorry, not possible.")
	return true
end

===== FILE_START: TOTAL/actions/scripts/Jakundaf Desert/The Pits of Inferno Quest/SkeletonReward.lua =====
-- [PROJECT 7.7 TFS 1.5] Converted script
-- Purpose: Skeleton reward
-- Notes: API updates, safety checks

function onUse(player, item, fromPosition, target, toPosition)
	if item.uid ~= 10282 then
		return true
	end

	local status = player:getStorageValue(10282)

	if status <= 0 then
		local bp = player:addItem(1988, 1) -- Pirate Backpack
		if bp then
			bp:addItem(2168, 1) -- Life Ring
			bp:addItem(2143, 5) -- White Pearls
			bp:addItem(2152, 11) -- Platinum Coins
		end

		player:setStorageValue(10282, 1)
		fromPosition:sendMagicEffect(12)
		player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "You have found something in the skeleton.")
		return true
	end

	player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The corpse is empty.")
	return true
end
