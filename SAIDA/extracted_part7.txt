-- [PROJECT 7.7 TFS 1.5] Converted script
-- Purpose: Gate + teleport lever
-- Notes: melhorias, atualizações da API, otimizações
-- Changes: English messages, gate safety, teleport cleanup, flow fixed

function onUse(player, item, fromPosition, target, toPosition)
local gatePos = Position(33314, 31592, 15)
local telePos = Position(33316, 31591, 15)
local goPos   = Position(33321, 31591, 14)

local gateItem = Tile(gatePos):getItemById(1355) -- stone wall
local teleportItem = Tile(telePos):getItemById(1387) -- teleport

-- Lever ON
if item.uid == 10002 and item.itemid == 1945 then
    if gateItem then gateItem:remove() end
    item:transform(1946)
    Game.createTeleport(1387, goPos, telePos)
    return true
end

-- Lever OFF
if item.uid == 10002 and item.itemid == 1946 then
    if not gateItem then
        Game.createItem(1355, 1, gatePos)
    end
    item:transform(1945)
    if teleportItem then teleportItem:remove() end
    return true
end

player:sendCancelMessage("Sorry, not possible.")
return true
end

===== FILE_START: TOTAL/actions/scripts/Edron/The Annihilator Quest/annihilator_chest.lua =====
#<ABSOLUTE_PATH>/home/maicon/Documentos/GitHub/DATAPACK--7.7/TOTAL/actions/scripts/Edron/The Annihilator Quest/annihilator_chest.lua</ABSOLUTE_PATH>
-- [PROJECT 7.7 TFS 1.5] Converted script
-- Purpose: Annihilator multi-chest system
-- Notes: melhorias, atualizações da API, otimizações
-- Changes: fixed storage, English messages, reward unification, item comments added

function onUse(player, item, fromPosition, target, toPosition)
local storage = 10010

if player:getGroup():getAccess() then
  return true
end

local rewardMsg = ""
local rewardId = nil
local extraContainerItem = nil

if item.uid == 10010 then
  rewardMsg = "You have found a Demon Armor."
  rewardId = 2494 -- Demon Armor

elseif item.uid == 10011 then
  rewardMsg = "You have found a Sword of Valor."
  rewardId = 2400 -- Sword of Valor

elseif item.uid == 10012 then
  rewardMsg = "You have found a Stonecutter's Axe."
  rewardId = 2431 -- Stonecutter's Axe

elseif item.uid == 10013 then
  rewardMsg = "You have found a Present."
  rewardId = 1990 -- Present
  extraContainerItem = 2326 -- inside item

else
  return false
end

if player:getStorageValue(storage) < 1 then
  player:sendTextMessage(MESSAGE_INFO_DESCR, rewardMsg)

  local reward = player:addItem(rewardId, 1)
  if extraContainerItem then
    reward:addItem(extraContainerItem, 1)
  end

  player:setStorageValue(storage, 1)
else
  player:sendTextMessage(MESSAGE_INFO_DESCR, "The chest is empty.")
end

return true
end
