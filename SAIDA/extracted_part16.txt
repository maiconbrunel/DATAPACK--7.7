-- [PROJECT 7.7 TFS 1.5] Converted script
-- Summary: Updated to API modern, fixed teleport logic, removed duplicate transforms, improved safety.

function onUse(player, item, fromPosition, target, toPosition)
	local playerPos = player:getPosition()
	local newPos = Position(368, 1386, 12)

	if item.uid == 10264 and item.itemid == 1945 then
		item:transform(1946)
		playerPos:sendMagicEffect(2)
		player:teleportTo(newPos)
		newPos:sendMagicEffect(10)
		return true
	end

	if item.uid == 10264 and item.itemid == 1946 then
		item:transform(1945)
		return true
	end

	player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "Sorry, not possible.")
	return true
end

-- [PROJECT 7.7 TFS 1.5] Converted script
-- Summary: Updated to modern API, validated storages, optimized conditions, improved effects & safety.

function onUse(player, item, fromPosition, target, toPosition)
	local pos = player:getPosition()

	if item.uid == 10281 and pos.x == 346 and (pos.y == 1419 or pos.y == 1418) and pos.z == 12 then
		if player:getStorageValue(10274) <= 0
		or player:getStorageValue(10275) <= 0
		or player:getStorageValue(10276) <= 0
		or player:getStorageValue(10277) <= 0
		or player:getStorageValue(10278) <= 0
		or player:getStorageValue(10279) <= 0
		or player:getStorageValue(10280) <= 0 then
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "You did not absorb enough energy!")
			return true
		end

		local newPos = Position(344, 1419, 12)
		player:teleportTo(newPos)
		newPos:sendMagicEffect(10)
		return true
	end

	if item.uid == 10281 and pos.x == 344 and (pos.y == 1419 or pos.y == 1420) and pos.z == 12 then
		local newPos = Position(346, 1419, 12)
		pos:sendMagicEffect(2)
		player:teleportTo(newPos)
		newPos:sendMagicEffect(10)
		return true
	end

	return true
end
