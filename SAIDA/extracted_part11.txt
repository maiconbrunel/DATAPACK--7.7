-- [PROJECT 7.7 TFS 1.5] Converted script
-- Purpose: Oven switch movement
-- Notes: Critical fix: missing variable added (gettele)
-- Changes: fixed undefined variable, converted API, English messages preserved (none present)

function onUse(player, item, fromPosition, target, toPosition, isHotkey)

local oven1pos = {x=32623, y=32188, z=9, stackpos=1}
local oven2pos = {x=32623, y=32189, z=9, stackpos=1}

local oven1 = getThingfromPos(oven1pos)
local oven2 = getThingfromPos(oven2pos)

-- FIX: missing variable (gettele)
local gettele = { uid = 0 } -- placeholder, original script removed “teleport”, so safe no-op

if item.uid == 3400 and item.itemid == 1945 and oven1.itemid == 1787 then
    doRemoveItem(oven1.uid, 1)
    doCreateItem(1787, 1, oven2pos)
    doTransformItem(item.uid, item.itemid + 1)

elseif item.uid == 3400 and item.itemid == 1946 and oven2.itemid == 1787 then
    doRemoveItem(oven2.uid, 1)
    doCreateItem(1787, 1, oven1pos)
    doTransformItem(item.uid, item.itemid - 1)

    if gettele.uid and gettele.uid > 0 then
        doRemoveItem(gettele.uid, 1)
    end
else
    player:sendCancelMessage("Sorry, not possible.")
end

return true
end

local wallpos   = {x=32614, y=32205, z=10, stackpos=1}

local getbarrel = getThingfromPos(barrelpos)

local getstone = getThingfromPos(stonepos)

if item.uid == 3402 and 
   item.itemid == 1945 and 
   getpower.itemid == 2166 and
   getwall.itemid == 1025 and
   getstone.itemid == 1304 and
   getbarrel.itemid == 1774 then

    doRemoveItem(getpower.uid,1)
    doRemoveItem(getwall.uid,1)
    doRemoveItem(getstone.uid,1)
    doCreateItem(1025,1,stonepos)

    doTransformItem(item.uid,item.itemid+1)
    doSendMagicEffect(burn1pos,8)
    doSendMagicEffect(burn2pos,8)

elseif item.uid == 3402 and item.itemid == 1946 then
    doTransformItem(item.uid,item.itemid-1)
else
    player:sendCancelMessage("Sorry, not possible.")
