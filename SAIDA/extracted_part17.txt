-- [PROJECT 7.7 TFS 1.5] Converted script
-- Purpose: POI reward chests
-- Notes: API updates, storage checks, English messages

function onUse(player, item, fromPosition, target, toPosition)
	local uid = item.uid
	local storage = player:getStorageValue(uid)

	local function giveReward(text, itemId, count)
		player:sendTextMessage(MESSAGE_EVENT_ADVANCE, text)
		player:addItem(itemId, count)
		player:setStorageValue(uid, 1)
	end

	if uid == 10544 then -- Arcane Staff
		if storage <= 0 then
			giveReward("You have chosen an arcane staff.", 2453, 1)
		else
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
		end

	elseif uid == 10545 then -- Epee
		if player:getStorageValue(10544) <= 0 then
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "You have chosen an epee.")
			player:addItem(2438, 1)
			player:setStorageValue(10544, 1)
		else
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
		end

	elseif uid == 10546 then -- Crystal Arrow
		if player:getStorageValue(10544) <= 0 then
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "You have chosen a crystal arrow.")
			player:addItem(2352, 1)
			player:setStorageValue(10544, 1)
		else
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
		end

	elseif uid == 10547 then -- Soft Boots
		if storage <= 0 then
			giveReward("You have found a pair of soft boots.", 2358, 1)
		else
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
		end

	elseif uid == 10548 then -- Backpack of Holding
		if storage <= 0 then
			player:setStorageValue(uid, 1)
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "You have found a backpack.")
			local bp = player:addItem(1988, 1)
			local ids = {2273, 2268, 2313, 2311}
			local charges = {10, 10, 30, 50}
			for i = 1, #ids do
				bp:addItem(ids[i], charges[i])
			end
		else
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
		end

	elseif uid == 10549 then -- Panda Bear
		if storage <= 0 then
			giveReward("You have found a panda bear.", 5080, 1)
		else
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
		end

	elseif uid == 10550 then -- Frozen Starlight
		if storage <= 0 then
			giveReward("You have found a frozen starlight.", 2361, 1)
		else
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
		end

	elseif uid == 10551 then -- Ornamented Ankh
		if storage <= 0 then
			giveReward("You have found an ornamented ankh.", 2354, 1)
		else
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
		end

	elseif uid == 10552 then -- Holy Tible
		if storage <= 0 then
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "You have found the Holy Tible.")
			local book = player:addItem(1970, 1)
			book:setAttribute(ITEM_ATTRIBUTE_TEXT,
				"Banor I praise your name.\n" ..
				"Be with me in the battle.\n" ..
				"Be my shield, let me be your sword.\n" ..
				"I will honour the godly spark in my soul. May it flourish and grow.")
			player:setStorageValue(10552, 1)
		else
			player:sendTextMessage(MESSAGE_EVENT_ADVANCE, "The chest is empty.")
		end
	end

	return true
end

-- [PROJECT 7.7 TFS 1.5] Converted script
-- Purpose: POH gate switch
-- Notes: API updates, Tile/Position conversion

function onUse(player, item, fromPosition, target, toPosition)
	local gatePos = Position(32780, 32231, 8)
	local tile = Tile(gatePos)
	local gate = tile and tile:getItemById(387)

	if item.uid == 10105 and item.itemid == 1945 and gate then
		gate:remove()
		item:transform(1946)
		return true

	elseif item.uid == 10105 and item.itemid == 1946 and not gate then
		Game.createItem(387, 1, gatePos)
		item:transform(1945)
		return true
	end

	player:sendCancelMessage("Sorry, not possible.")
	return true
end
